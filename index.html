const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const scoreEl = document.getElementById('score'); const winScreen = document.getElementById('win-screen'); canvas.width = window.innerWidth; canvas.height = window.innerHeight; let score = 0; let gameOver = false; const items = []; const basket = { x: canvas.width / 2 - 50, y: canvas.height - 100, w: 100, h: 60 }; const wishes = ["🌹", "🌸", "💐", "💖", "✨"]; function moveHandler(e) { if (gameOver) return; let clientX = e.touches ? e.touches[0].clientX : e.clientX; basket.x = clientX - basket.w / 2; if (basket.x < 0) basket.x = 0; if (basket.x > canvas.width - basket.w) basket.x = canvas.width - basket.w; } canvas.addEventListener('touchmove', moveHandler); canvas.addEventListener('mousemove', moveHandler); function createItem() { if (gameOver) return; items.push({ x: Math.random() * (canvas.width - 40), y: -50, s: 3 + Math.random() * 4, type: Math.random() > 0.3 ? 'GIFT' : 'FLOWER', icon: wishes[Math.floor(Math.random() * wishes.length)] }); } function drawBasket() { ctx.fillStyle = "#8b4513"; ctx.beginPath(); ctx.moveTo(basket.x, basket.y); ctx.lineTo(basket.x + basket.w, basket.y); ctx.lineTo(basket.x + basket.w - 15, basket.y + basket.h); ctx.lineTo(basket.x + 15, basket.y + basket.h); ctx.closePath(); ctx.fill(); ctx.fillStyle = "white"; ctx.font = "14px Arial"; ctx.fillText("CLASS 8A", basket.x + 18, basket.y + 35); } function update() { if (gameOver) return; ctx.clearRect(0, 0, canvas.width, canvas.height); drawBasket(); for (let i = items.length - 1; i >= 0; i--) { let it = items[i]; it.y += it.s; ctx.font = "35px Arial"; ctx.fillText(it.type === 'GIFT' ? "🎁" : it.icon, it.x, it.y); if (it.y > basket.y && it.y < basket.y + basket.h && it.x > basket.x - 20 && it.x < basket.x + basket.w) { score++; scoreEl.innerText = score; items.splice(i, 1); // Cài đặt các mốc thoại theo yêu cầu của Hoàng Minh if (score === 10) { alert("Cô còn đang non nhoá!"); } else if (score === 20) { alert("Nếu mà chơi đến đây thì cô chứ cs thưởng mô"); } else if (score === 50) { alert("Phần quà của cô là iPhone 17 Pro Max nhá nhưng chơi đến 100 điểm sẽ cs bất ngờ"); } else if (score === 100) { alert("Ôi thôi check, e đã đc 1 căn villa ràu thôi đừng chơi nữa"); gameOver = true; // Dừng trò chơi lại winScreen.style.display = 'block'; } } else if (it.y > canvas.height) { items.splice(i, 1); } } requestAnimationFrame(update); } setInterval(createItem, 800); update();
